Just change the way you init your tables:

$(document).ready(function() {
  $('table.dataTable').each(function(i) {
    var thisTable = $(this);
    var thisName = thisTable.attr('name');
    thisTable.DataTable({
      dom: 'Bfrtip',
      buttons: [{
        extend: "excelHtml5",
        text: "Export to XLSx",
        filename: function() {
          filename = thisName;
          return filename;
        }
      }]
    });
  })
});
