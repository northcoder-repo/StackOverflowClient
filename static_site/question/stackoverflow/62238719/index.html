The following example is based on the demo here.

This is a single row grouping (not a multi-row grouping) since the example data in the question only has a single group.

It uses the following standard scripts and CSS files, with the addition of two "row group" plug-in files:

<head>
  <meta charset="utf-8">
  <title>Demo</title>
  <script src="https://code.jquery.com/jquery-3.5.1.min.js"></script>
  <script src="https://cdn.datatables.net/1.10.21/js/jquery.dataTables.min.js"></script>
  <link rel="stylesheet" type="text/css" href="https://cdn.datatables.net/1.10.21/css/jquery.dataTables.min.css">
  <!-- this next one is optional: -->
  <link rel="stylesheet" type="text/css" href="https://datatables.net/media/css/site-examples.css">

  <!-- row group plugin -->
  <link href="https://cdn.datatables.net/rowgroup/1.0.2/css/rowGroup.dataTables.min.css" rel="stylesheet" type="text/css" />
  <script src="https://cdn.datatables.net/rowgroup/1.0.2/js/dataTables.rowGroup.min.js"></script>

</head>

The DataTables initialization code is as follows:

<script type="text/javascript">

var collapsedGroups = {};
var dtable;

$(document).ready(function() {
    $('#example').DataTable( {
        "pageLength": 50,
        rowGroup: {
            dataSrc: 0,

            startRender: function(rows, group) {
                var collapsed = !!collapsedGroups[group];
                rows.nodes().each(function (r) {
                    r.style.display = collapsed ? 'none' : '';
                });    

                var toggleClass = collapsed ? 'fas fa-chevron-up' : 'fas fa-chevron-down';

                // Add category name to the <tr>. NOTE: Hardcoded colspan
                return $('<tr/>')
                    .append('<td colspan="' + rows.columns()[0].length + 
                            '"><i class="'+ toggleClass + '"></i> ' + 
                            group + ' (' + rows.count() + ')</td>')
                    .attr('data-name', group)
                    .toggleClass('collapsed', collapsed);
            }

        }

    } );
} );

</script>
